// The Web App URL from Step 3
const scriptURL = "YOUR_WEB_APP_URL_HERE";

function submitTicket() {
  
  // 1. Gather data
  const formData = {
    fullName: document.getElementById("nameInput").value,
    email: document.getElementById("emailInput").value,
    category: document.getElementById("categorySelect").value, // Dropdown value
    subject: document.getElementById("subjectInput").value,
    message: document.getElementById("messageInput").value
  };

  // 2. Basic Validation
  if(!formData.fullName || !formData.email || !formData.subject || !formData.message) {
    alert("Please fill in all required fields.");
    return;
  }

  // 3. Send to Google Sheet
  const submitButton = document.getElementById("submitBtn");
  submitButton.innerText = "Sending..."; // Optional: feedback to user
  
  fetch(scriptURL, {
    method: "POST",
    body: JSON.stringify(formData),
    mode: "no-cors" // keeps the request simple
  })
  .then(() => {
    alert("Ticket Submitted Successfully!");
    submitButton.innerText = "Submit Ticket";
    // Optional: clear the form
    // document.getElementById("ticketForm").reset();
  })
  .catch(error => {
    console.error("Error!", error);
    alert("Error submitting ticket.");
    submitButton.innerText = "Submit Ticket";
  });
}
